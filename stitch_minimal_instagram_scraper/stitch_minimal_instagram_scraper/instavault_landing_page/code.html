import { InstaVault } from '@instavault/core';


const client = new InstaVault({

  apiKey: process.env.INSTAVAULT_KEY,

  session: 'persist'

});


const data = await client.profile('@example').posts({

  limit: 500,

  include_metrics: true

});


console.log(`Extracted ${data.length} entries.`);
